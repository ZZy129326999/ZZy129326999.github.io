# 迭代加深搜索

在暴力枚举中，著名的 IDA∗IDA^*IDA∗，又称为迭代加深搜索，是一种递增迭代最大深度 maxDeepmaxDeepmaxDeep 的枚举方式，这是迭代加深的含义，在此基础上 IDA∗IDA^*IDA∗ 往往满足一个重要的剪枝条件，即

maxDeep−curDeep+1<=需要匹配的剩余价值/当前迭代的价值maxDeep - curDeep + 1 <= 需要匹配的剩余价值/当前迭代的价值maxDeep−curDeep+1<=需要匹配的剩余价值/当前迭代的价值

# 题目列表

  1. UVA12558 埃及分数

思路：

题目要求出 a/ba/ba/b 被若干 1/ck1/c_k1/ck​ 的形式，其中还有一些数字是规定 ckc_kck​ 不能取到的。很明显没有显而易见的定理和规律佐证，所以本题只能使用深搜去枚举 ckc_kck​ 的所有可能取值。

枚举方式：可以看出，如果一直按照数字枚举，是没有固定的最大深度的，因为一般的深搜都是有最大深度的限制，所以，每次枚举最大深度，以此使用深搜，问题就可以解决了，此外还要注意数字的枚举技巧，比如给出 a/ba/ba/b，怎么才能知道从哪个数字开始呢？题目规定分子必须是 1， ckc_kck​ 必须是整数，所以先把 ab\frac{a}{b}ba​ 化简成 1b/a\frac{1}{b/a}b/a1​，C++ 语法中会对 b/ab/ab/a 向下取整，但是会造成数字比剩余和 a/ba/ba/b 要大，所以从 1b/a+1\frac{1}{b/a+1}b/a+11​ 枚举。

  • 使用 vis 数组记录不能使用的数字时，需要判定是否小于 1000，因为枚举的广度可能很大，会超过 1000，造成数组越界。
  • 记录答案的数组要开 long long ！！！！
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
const int N = 1e3+7;
ll vis[N], v[N], ans[N], maxd = 1;
ll get_first(ll a, ll b) {
    return b/a + 1;
}
ll gcd(ll a, ll b) {
    return b ? gcd(b, a%b) : a; 
}
bool better(int d) {
    for (int i = d; i >= 0; i--) {
        if (v[i] != ans[i]) {
            return ans[i] == -1 || v[i] < ans[i];
        }
    }    
    return false;
}
bool dfs(int d, ll a, ll b, ll c) {
    if (d == maxd) {
        if (a>1 || (b <= 1000 && vis[b])) return false;
        v[d] = b;
        // printf("%d, %lld\n", d, a);
        if (better(d)) memcpy(ans, v, sizeof(ll)*(d+1));
        return true;
    }
    c = max(c, get_first(a, b));
    bool ok = false;
    for (ll i = c; ; i++) {
        if ((maxd-d+1)*b <= a*i) break;
        if (i <= 1000 && vis[i]) continue;
        v[d] = i;
        // printf("%d, %lld\n", d, i);
        // a/b - 1/i
        ll aa = a*i - b, bb = b*i;
        ll g = gcd(aa, bb);
        if (dfs(d+1, aa/g, bb/g, i+1)) ok = true;
    }
    return ok;
}
int main() {
    int T, k, cnt = 0;
    ll a, b;
    scanf("%d", &T);
    while (T--) {
        memset(vis, 0, sizeof(vis));
        scanf("%lld%lld%d", &a, &b, &k); 
        for (int i = 0, x; i < k; i++) scanf("%d", &x), vis[x] = 1;
        for (maxd = 1; ; maxd++) {
            memset(ans, -1, sizeof(ans));
            if (dfs(0, a, b, get_first(a, b))) {
                // printf("maxd=%d", maxd);
                break;
            }
        }
        printf("Case %d: %lld/%lld=1/%lld", ++cnt, a, b, ans[0]);
        for (int i = 1; i <= maxd; i++) {
            printf("+1/%lld", ans[i]);
        } 
        puts("");
    }
    return 0;
}
